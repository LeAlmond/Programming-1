                                printf("You have exceed your max withdrawl Amount\n");
                        }
                        break;
                case 2:
                        printf("Please Enter an Amount to Deposit\n");
                        scanf("%f", &trans);
                        balance = balance + trans;
                        printf("You Have Deposited $%.2f\n",trans);
                        break;
